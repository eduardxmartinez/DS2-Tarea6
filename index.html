package ds2.tarea6;

import java.util.Scanner;

public class DS2Tarea6 {

    public static void main(String args[]) {
        int z;
        sumar();
        desplegar(9);
        z = cuadrado(80);
        System.out.println(z);
        raizCuadrada();
        mayor();
        mayorArray();
    }

    public static void sumar() {
        System.out.println("\n****METODO SUMAR****");
        int n1, n2, suma;
        Scanner sc = new Scanner(System.in);
        System.out.print("Valor 1 : ");
        n1 = sc.nextInt();
        System.out.print("Valor 2 : ");
        n2 = sc.nextInt();
        suma = n1 + n2;
        System.out.println("La suma es : " + suma);
    }

    public static void desplegar(int x) {
        System.out.println("\n****METODO DESPLEGAR****");
        int i;
        for (i = 1; i <= x; i++) {
            System.out.print(i + " ");
        }
        System.out.println("");
    }

    public static int cuadrado(int x) {
        System.out.println("\n****METODO CUADRADO****");
        int resultado;
        resultado = x * x;
        return resultado;
    }

    public static void raizCuadrada() {
        System.out.println("\n****METODO RAIZ CUADRADA****");
        double resultado, n;
        Scanner sc = new Scanner(System.in);
        System.out.print("Ingresa un valor: ");
        n = sc.nextDouble();
        resultado = Math.pow(n, 0.5);
        System.out.println("La raíz cuadrada es: " + resultado);
    }

    public static void mayor() {
        System.out.println("\n****METODO MAYOR DE 3 NUMEROS****");
        double[] array = new double[3];
        double mayor = 0;
        Scanner sc = new Scanner(System.in);
        System.out.println("Ingrese 3 números");
        for (int i = 0; i < 3; i++) {
            System.out.print("Valor " + (i + 1) + ": ");
            array[i] = sc.nextDouble();
            mayor = array[0];
            if (array[i] >= mayor) {
                mayor = array[i];
            }
        }
        System.out.println("El número mayor es: " + mayor);
    }

    public static void mayorArray() {
        System.out.println("\n****METODO MAYOR DE UN ARREGLO****");
        double[][] array = new double[4][4];
        double mayor = 0;
        Scanner sc = new Scanner(System.in);
        System.out.println("Ingrese 16 números");
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                System.out.print("Valor [" + i + "][" + j + "]: ");
                array[i][j] = sc.nextDouble();
            }
        }
        mayor = array[0][0];
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                if (array[i][j] >= mayor) {
                    mayor = array[i][j];
                }
            }
        }
        System.out.println("El número mayor es: " + mayor);
    }

}
